n = int(input())
for length in range(n):
    print(' '*(n-length-1)+'#'*(length+1))

#include<iostream>
using namespace std;
int main()
{
    int a[10],b[10],e=0,d=0;
    for(int i=0;i<3;i++)
    {
        cin>>a[i];
        cin>>b[i];}
        for(int i=0;i<3;i++)
        {
        if(a[i]>b[i])
            e=e+1;
        else if(a[i]<b[i])
            d=d+1; 
        else
            e=e+0;d=d+0;
        
    }
    cout<<e<<" "<<d;
}
