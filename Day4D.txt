Picking Numbers

#include <iostream>
using namespace std;

int n,a[100],b[100];
int main() {
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            int m= 2*i;
            if(a[i]==a[j]||a[i]==a[j]+1)
                b[m]++;
            if(a[i]==a[j]||a[i]==a[j]-1)
                b[m+1]++;
        }
    }
    for(int i=0;i<200;i++)
        if(b[i]>b[0])
            b[0]=b[i];
    cout<<b[0];
    return 0;
}

Counting Valleys

#include <iostream>
using namespace std;

int main()
{
    unsigned int n;
    cin>>n;
    if(n>=2 && n<=1000000)
    {
        char S[n];
        int level=0, cnt=0;
        for(unsigned int i=0; i<n; ++i)
        {
            cin>>S[i]; 
            if(S[i]=='U')
                ++level;
            if(S[i]=='D')
                --level;
            if(level==0 && S[i]=='U')
                ++cnt;
        }
    cout<<cnt<<endl;
    }
    return 0;
}
